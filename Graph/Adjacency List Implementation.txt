#include <iostream>
#include <unordered_map>
#include <list>
using namespace std;

class Graph{
    
    public:
        unordered_map<int,list<int>> adj;
        
        void addEdge(int u,int v,bool dir){
            adj[u].push_back(v);
            
            if(dir == 0)
                adj[v].push_back(u);
        }
        
        void printAdjList(){
            for(auto i:adj){
                cout<<i.first<<"->";
                for(auto j:i.second)
                    cout<<j<<",";
                cout<<endl;
            }
        }
};


int main() {
    
    int n;
    cout<<"Enter the no. of nodes"<<endl;
    cin>>n;
    
    int e;
    cout<<"Enter the no. of edges"<<endl;
    cin>>e;
    
    Graph g;
    
    for(int i =0;i<e;i++){
        int u,v;
        cin>>u>>v;
        g.addEdge(u,v,0);
    }
    
    g.printAdjList();
    
    return 0;
}
